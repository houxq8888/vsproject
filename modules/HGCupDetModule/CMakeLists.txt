    HGCommon
    HGImageAlgorithmService
    ${ZXING_LIBRARY}
    ${OpenCV_LIBRARIES}
    ${LOG4CPLUS_LIBRARIES}  
)
if (PLATFORM STREQUAL "x64")

HG_CMAKE_MODULE_DEPEND(
    HGCommon
    HGImageAlgorithmService
    ${ZXING_LIBRARY}
    ${OpenCV_LIBRARIES}
    ${LOG4CPLUS_LIBRARIES}  
)

    HGCommon
    HGImageAlgorithmService
